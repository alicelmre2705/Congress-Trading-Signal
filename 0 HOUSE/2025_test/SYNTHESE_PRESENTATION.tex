\section{Le pipeline en détail}
% =====================================================================

\subsection*{En amont (commun aux deux étapes)}
\begin{enumerate}
  \item Construire le référentiel des élus (bioguide, parti, comités clés) $\to$
        \file{01\_ref\_universe.csv} (12\,767) et \file{02\_ref\_house\_key.csv} (126 ; comités
        Financial / Armed / Intelligence).
  \item Télécharger l'index des PTR et filtrer la fenêtre T1 2025 $\to$ \file{03\_ptr\_index.csv} (117).
  \item Télécharger chaque PDF, tester s'il contient du texte (\texttt{has\_text}), et le trier :
        \textbf{lisible} ou \textbf{scanné} $\to$ \file{04\_download\_manifest.csv} (117 $\to$ 100 / 17).
\end{enumerate}

\subsection{Étape 1 --- PDF électroniques (lisibles)}
\textbf{100 PDF} contenant du texte sélectionnable.
\begin{itemize}
  \item Extraction du texte (\texttt{pdfplumber}).
  \item Parsing par expressions régulières : type d'opération, dates, ticker, montant, propriétaire.
  \item Gestion des lignes coupées sur plusieurs lignes (continuation).
  \item Jointure à l'identité du déposant (bioguide).
\end{itemize}
\textbf{Résultat :} \file{06\_house\_2025q1\_transactions.csv} --- \nb{1\,105 transactions}.
\textbf{0 échec} de parsing (\file{05\_parse\_failures.csv} vide).

\subsection{Étape 2 --- PDF scannés (OCR), enrichissement et fusion}
\textbf{17 PDF} sans texte (images), lus par \textbf{Claude Vision} (\texttt{claude-sonnet-4-6}).
\begin{itemize}
  \item Chaque PDF $\to$ images (DPI 200) $\to$ lecture par lots ; \textbf{sortie structurée forcée}
        (\texttt{tool\_use} sur \texttt{record\_transactions}, champs validés) --- pas de JSON fragile.
  \item \textbf{Cache versionné} (par \texttt{doc\_id} + empreinte du prompt) : un re-run sans
        changement ne rappelle pas l'API (gratuit).
  \item \textbf{Filtre de la ligne-exemple} pré-imprimée sur les formulaires (« Example Mega Corp »),
        écartée systématiquement.
  \item \textbf{Enrichissement \texttt{ticker} en 3 passes tracées} (\texttt{ticker\_source}) :
        \texttt{explicit} (présent dans le nom), \texttt{elec\_dict} (dictionnaire bâti sur les lignes
        électroniques tickées), puis \textbf{passe LLM} (Claude nom$\to$ticker, sortie structurée,
        cache versionné \file{ticker\_llm\_cache.json} ; \nb{228 / 310} noms résolus) ; plus
        \texttt{asset\_type} par règles.
  \item \textbf{Déduplication inter-sources uniquement} : les répétitions réelles intra-OCR
        (vrais trades répétés) sont conservées.
\end{itemize}
\textbf{Résultats :} \file{06b\_house\_2025q1\_ocr\_transactions.csv} (1\,167) puis \textbf{fusion}
$\to$ \file{06\_house\_2025q1\_transactions\_FINAL.csv} --- \nb{2\,272 transactions}.
\textbf{0 échec} de lot (\file{06c\_ocr\_failures.csv} vide) ; packaging \file{house\_2025q1\_FINAL.xlsx}.

\boite{%
\textbf{\color{accent}La passe LLM ticker, en bref.} Le dictionnaire électronique plafonnait la
couverture ticker OCR à 46\,\% (il ne connaît que les noms déjà tradés en électronique). La passe LLM
résout les grandes valeurs cotées absentes du dictionnaire $\to$ \textbf{OCR 46\,\% $\to$ 90\,\%},
auditée à \textbf{93\,\%} contre Quiver. Quiver n'est jamais utilisé pour remplir le ticker.}

\section{Limites connues (honnêteté)}
\label{sec:limites}
% =====================================================================

\begin{itemize}
  \item \textbf{Gonzalez --- 1 PTR sur 2 manqué.} Il a déposé \textbf{deux} PTR scannés. Le doc
        \texttt{8220796} est bien capté (3 transactions : HSBC, et TESLA acheté les 01/03 et 12/03).
        Le doc \texttt{8220809} est un \textbf{formulaire très pâle} : le modèle n'a lu que la
        \emph{ligne-exemple} (filtrée) $\to$ \textbf{0 ligne} extraite, sa transaction réelle (vente
        TESLA $\approx$ 17/03) est \textbf{manquée}. Quiver = 0 sur ce déposant, donc pas de référence externe.
  \item \textbf{Bilirakis.} 1 ligne (« GE Vernova », doc \texttt{8220747}) avec montant et date
        illisibles. Conservée mais incomplète --- \textbf{c'est le seul drapeau QA} de tout le pipeline.
  \item \textbf{\texttt{asset\_type} non harmonisé.} Le champ est rempli à 97,2\,\%, mais
        \textbf{hétérogène entre sources} : côté électronique, $\approx$ 39 lignes portent les
        \emph{codes bruts} du formulaire (\texttt{CT} = crypto comme Ethereum / mèmecoins,
        \texttt{CS} = notes corporate ou titres adossés, \texttt{HN} / \texttt{PS} / \texttt{OL}),
        non normalisés vers les libellés (\texttt{Stock}, \texttt{Mutual Fund}, \dots).
  \item \textbf{Ticker OCR résiduel ($\approx$ 10\,\%).} Les noms laissés vides sont des actifs
        \textbf{non cotés} (produits structurés, perpétuelles, options) ou des noms trop dégradés par
        l'OCR --- laissés vides à dessein, jamais devinés.
  \item \textbf{47 doublons exacts (Khanna), conservés.} Ses déclarations couvrent plusieurs
        \textbf{trusts familiaux} (Ahuja/Khanna) passant le même ordre le même jour, tous codés sous le
        même propriétaire $\to$ la même clé apparaît légitimement N fois (réel, confirmé page à page :
        Salesforce 27/02 ×4). Dédupliquer serait \textbf{destructeur} ; une minorité d'artefacts OCR
        subsistent, non séparables automatiquement ($\approx$ 0,1--0,5\,\%, confinés à Khanna).
\end{itemize}
