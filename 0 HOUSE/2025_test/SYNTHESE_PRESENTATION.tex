\section{Le pipeline en 2 étapes}
% =====================================================================

\subsection*{En amont (commun aux deux étapes)}
\begin{enumerate}
  \item Construire le référentiel des élus (bioguide, parti, comités clés) $\to$
        \file{01\_ref\_universe.csv} (12\,767) et \file{02\_ref\_house\_key.csv} (126).
  \item Télécharger l'index des PTR et filtrer la fenêtre T1 2025 $\to$ \file{03\_ptr\_index.csv} (117).
  \item Télécharger chaque PDF, tester s'il contient du texte (\texttt{has\_text}), et le trier :
        \textbf{lisible} ou \textbf{scanné} $\to$ \file{04\_download\_manifest.csv} (117).
\end{enumerate}

\subsection{Étape 1 --- PDF électroniques (lisibles)}
\textbf{100 PDF} contenant du texte sélectionnable.
\begin{itemize}
  \item Extraction du texte (\texttt{pdfplumber}).
  \item Parsing par expressions régulières : type d'opération, dates, ticker, montant, propriétaire.
  \item Gestion des lignes coupées sur plusieurs lignes (continuation).
  \item Jointure à l'identité du déposant (bioguide).
\end{itemize}
\textbf{Résultat :} \file{06\_house\_2025q1\_transactions.csv} --- \nb{1\,105 transactions}.
\textbf{0 échec} de parsing (\file{05\_parse\_failures.csv} vide).

\subsection{Étape 2 --- PDF scannés (OCR) + fusion}
\textbf{17 PDF} sans texte (images). Lus par \textbf{Claude Vision}.
\begin{itemize}
  \item Chaque PDF $\to$ images $\to$ lecture par lots ; \textbf{sortie structurée forcée}
        (\texttt{tool\_use}, champs validés) --- pas de JSON fragile dans du texte.
  \item \textbf{Cache versionné} (par \texttt{doc\_id} + empreinte du prompt) : un re-run sans
        changement ne rappelle pas l'API (gratuit).
  \item \textbf{Filtre de la ligne-exemple} pré-imprimée sur les formulaires (« Example Mega Corp »),
        écartée systématiquement.
  \item \textbf{Enrichissement} : \texttt{ticker} (depuis le nom, ou dictionnaire bâti sur l'étape 1)
        et \texttt{asset\_type} ; traçabilité via \texttt{ticker\_source}.
  \item \textbf{Déduplication inter-sources uniquement} : les répétitions réelles d'un même
        déposant (vrais trades répétés) sont conservées.
\end{itemize}
\textbf{Résultats :}
\file{06b\_house\_2025q1\_ocr\_transactions.csv} (1\,167) puis \textbf{fusion} $\to$
\file{06\_house\_2025q1\_transactions\_FINAL.csv} --- \nb{2\,272 transactions}.
\textbf{0 échec} de lot (\file{06c\_ocr\_failures.csv} vide).

\section{Validation contre QuiverQuant}
% =====================================================================

\textbf{Quiver, c'est quoi ?} QuiverQuant est une base \textbf{indépendante} qui recense les mêmes
transactions du Congrès. Elle sert ici d'\textbf{arbitre} : nos chiffres sont-ils justes et complets ?

\textbf{Méthode.} Comparaison \textbf{par déposant} (nous \emph{vs} Quiver). Fenêtre alignée sur la
\emph{date de divulgation} (sinon on comparerait des périodes différentes).

\midrule

\section{Limites connues (honnêteté)}
% =====================================================================

\begin{itemize}
  \item \textbf{Gonzalez} : 1 transaction (TESLA, vente du 17/03/25) \textbf{manquée} sur un
        formulaire très pâle. Quiver = 0 sur ce déposant, donc pas de référence externe.
  \item \textbf{Bilirakis} : 1 ligne (« GE Vernova ») avec montant/date illisibles. Conservée mais
        incomplète --- c'est le seul drapeau QA.
  \item \textbf{Ticker OCR plafonné à 46\,\%} : limite du dictionnaire électronique + fautes de
        lecture OCR sur certains noms.
  \item \textbf{Surplus Khanna} ($+280$ vs Quiver) : cohérent avec les contrôles visuels, mais
        \textbf{non audité ligne à ligne} (formulaire de 34 pages).
\end{itemize}
